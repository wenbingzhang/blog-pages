    3、关闭掉多余的终端 #

    RHEL5:/etc/inittab RHEL6:/etc/sysconfig/init

    4、关闭ipv6 #

    #vim /etc/sysconfig/network-script/ifcfg-eth0

    NETWORK_IPV6 = no

    #vim /etc/modprobe.d/ipv6.conf

    alias net-pf-10 off
alias ipv6 off
